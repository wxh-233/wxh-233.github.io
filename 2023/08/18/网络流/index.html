网络流

2023/8/18 图论

OI Wiki
P3376 网络最大流
P3376 网络最大流题解
本质就是求增广路的问题
求到无增广路时得到最大流
ps.增广路是指从s到t的一条路，流过这条路，使得当前的流（可以到达t的水量）可以增加。

网络流可以用来求解二分图最大匹配

错误代码示范

自写废物深搜代码，喜获37pts

#include<bits/stdc++.h>
#define int long long
using namespace std;
typedef int lsqxx;
struct lq{
    lsqxx v,w,nxt;
}e[10005];
lsqxx h[205],cnt=1;
void add(lsqxx u,lsqxx v,lsqxx w)
{
    e[++cnt].v=v;e[cnt].w=w;e[cnt].nxt=h[u];h[u]=cnt;
}
int n,m,s,td;
int x,y,z;
int ans;
int vis[205];
int dfs(int t,int cnt)
{
    if(t==td)
    {
        ans+=cnt;
        return cnt;
    }
    int flag=0;
    for(int i=h[t];i;i=e[i].nxt)
    {
        int v=e[i].v,w=e[i].w;
        if(vis[v]) continue;
        if(w<=0) continue;
        if(cnt==flag) break;
        vis[v]=1;
        int re=dfs(v,min(cnt-flag,w));
        e[i].w-=re;
        e[i^1].w+=re;
        flag+=re;
        vis[v]=0;
    }
    if(t==s)
        cout<<flag;
    return flag;
}
signed main()
{
    cin>>n>>m>>s>>td;
    vis[s]=1;
    for(int i=1;i<=m;i++)
        scanf("%lld%lld%lld",&x,&y,&z),
        add(x,y,z),add(y,x,0);
    dfs(s,0x7f7f7f7f);
    return 0;
}

超时原因1

RT.

如果搜的是 s->a->b->t 与 s->b->a->t 这两条边，就要搜两百次才能找到最大流
而运气好只要搜两次就能找到 ^rs

EK算法

如何使的搜的次数更小呢
引入EK算法
不使用深搜而采用广搜来找增广路
每次选取从源点到汇点最少边的增广路
直到选不到为止

复杂度 $O(nm^2)$ ~~明显适合稀疏图，但 $m$ 通常大于 $n$ ~~

正确代码

#include<bits/stdc++.h>
#define int long long
using namespace std;
typedef int lsqxx;
struct lq{
    lsqxx v,w,nxt;
}e[10005];
lsqxx h[205],cnt=1;
void add(int u,int v,int w)
{
    e[++cnt].v=v;e[cnt].w=w;e[cnt].nxt=h[u];h[u]=cnt;
}
int n,m,s,td;
int x,y,z;
int dis[205],vis[205],mis[205];
int bfs(int s,int td)
{
    queue<int>q;
    q.push(s);
    memset(dis,0,sizeof(dis));//存路径
    memset(vis,0,sizeof(vis));//找路径上最小边 
    memset(mis,0,sizeof(mis));//存cnt 
    vis[s]=0x7f7f7f7f;
    while(!q.empty())
    {
        int t=q.front();q.pop();
        for(int i=h[t];i;i=e[i].nxt)
        {
            int v=e[i].v,w=e[i].w;
            if(dis[v]) continue;
            if(w<=0) continue;
            dis[v]=t;
            vis[v]=min(vis[t],w);
            mis[v]=i;
            q.push(v);
            if(v==td)
                return vis[td];
        }
    }
    return -1;
}
void dfs(int t,int to)
{
    if(t==s) return;
    e[mis[t]].w-=to;
    e[mis[t]^1].w+=to;
    dfs(dis[t],to);
}
int EK(int s,int t) 
{
    int ans=0;
    while(1)
    {
        int cnt=bfs(s,t);
        if(cnt==-1) break;
        ans+=cnt;
        dfs(t,cnt);
    }
    return ans;
}
signed main()
{
    cin>>n>>m>>s>>td;
    for(int i=1;i<=m;i++)
        scanf("%lld%lld%lld",&x,&y,&z),add(x,y,z),add(y,x,0);
    cout<<EK(s,td);
    return 0;
}

ISAP

引入高度的概念
对于每一个点，开始的高度为 $0$ ，每个节点只能向高度比它低的节点流出，汇点高度固定为 $0$
每当流入的流量大于流出的流量，即由这个点无法流出，高度 $+1$
当源点的高度大于等于 $n$ 时，得到最大流，输出即可

纯代码，不带优化

//ISAP
#include<bits/stdc++.h>
#define int long long
using namespace std;
namespace lsq {
    typedef int lsqxx;
    struct lq {
        struct lqbz {
            lsqxx v,w,nxt;
        } e[1000005];
        lsqxx h[1000005],cnt=1;
        inline void add(lsqxx u,lsqxx v,lsqxx w=1) {
            e[++cnt].v=v;
            e[cnt].w=w;
            e[cnt].nxt=h[u];
            h[u]=cnt;
        }
    #define F(z,u) for(int j=z.h[u],v=z.e[j].v,w=z.e[j].w;j;j=z.e[j].nxt,v=z.e[j].v,w=z.e[j].w)
    }q;
};
using namespace lsq;
int n,m,s,t;
int x,y,z;
int d[205];
int dfs(int u=s,int in=INT_MAX)
{
    if(u==t) return in;
    int out=0;
    F(q,u)
    {
        if(d[u]>d[v]&&w)
        {
            int nt=dfs(v,min(w,in));
            in-=nt;out+=nt;
            q.e[j].w-=nt;q.e[j^1].w+=nt;
            if(!in) return out;
        }
    }
    ++d[u];
    return out;
}
int ISAP()
{
    int ans=0;
    while(d[s]<n)
        ans+=dfs();
    return ans;
}
signed main()
{
    cin>>n>>m>>s>>t;
    for(int i=1;i<=m;i++)
    {
        scanf("%lld%lld%lld",&x,&y,&z);
        if(x==y) continue;
        q.add(x,y,z),q.add(y,x,0);
    }
    cout<<ISAP();
    return 0;
}

$BFS$ 优化

//ISAP
#include<bits/stdc++.h>
#define int long long
using namespace std;
namespace lsq {
    typedef int lsqxx;
    struct lq {
        struct lqbz {
            lsqxx v,w,nxt;
        } e[1000005];
        lsqxx h[1000005],cnt=1;
        inline void add(lsqxx u,lsqxx v,lsqxx w=1) {
            e[++cnt].v=v;
            e[cnt].w=w;
            e[cnt].nxt=h[u];
            h[u]=cnt;
        }
    #define F(z,u) for(int j=z.h[u],v=z.e[j].v,w=z.e[j].w;j;j=z.e[j].nxt,v=z.e[j].v,w=z.e[j].w)
    }q,p;
};using namespace lsq;
int n,m,s,t;
int x,y,z;
int d[205];
void BFS()
{
    memset(d,-1,sizeof(d));
    queue<int>r;
    r.push(t);
    d[t]=0;
    while(!r.empty())
    {
        int cht=r.front();r.pop();
        F(p,cht)
        {
            if(d[v]!=-1) continue;
            d[v]=d[cht]+1;
            r.push(v);
        }
    }
}
int dfs(int u=s,int in=INT_MAX)
{
    if(u==t) return in;
    int out=0;
    F(q,u)
    {
        if(d[u]>d[v]&&w)
        {
            int nt=dfs(v,min(w,in));
            in-=nt;out+=nt;
            q.e[j].w-=nt;q.e[j^1].w+=nt;
            if(!in) return out;
        }
    }
    ++d[u];
    return out;
}
int ISAP()
{
    int ans=0;
    BFS();
    while(d[s]<n)
        ans+=dfs();
    return ans;
}
signed main()
{
    cin>>n>>m>>s>>t;
    for(int i=1;i<=m;i++)
    {
        int _=scanf("%lld%lld%lld",&x,&y,&z);
        if(x==y) continue;
        q.add(x,y,z),q.add(y,x,0),p.add(y,x,z);
    }
    cout<<ISAP();
    return 0;
}
